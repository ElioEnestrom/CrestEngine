#version 330 core

struct Material {
    sampler2D diffuse;
    sampler2D specular;
    float shininess;
}; 
struct Light {
    vec3 direction;
  
    vec3 ambient;
    vec3 diffuse;
    vec3 specular;
};

  
out vec4 FragColor;

in vec2 TexCoord;
in vec3 FragPos;
in vec3 Normal;

uniform Light light; 
uniform Material material;
uniform vec3 lightPos;
uniform vec3 objectColor;
uniform vec3 lightColor;
uniform vec3 viewPos;
uniform bool viewSpecular;

uniform sampler2D texture1;
uniform sampler2D texture2;
uniform float textureMixer;

void main()
{
    // Mix the two textures using the textureMixer uniform
    vec4 mixedTexture = mix(texture(texture1, TexCoord), texture(texture2, TexCoord), textureMixer);

    // Ambient
    vec3 ambient = vec3(mixedTexture) * light.ambient;

    // Diffuse
    vec3 norm = normalize(Normal);
    vec3 lightDir = normalize(-light.direction);
    float diff = max(dot(norm, lightDir), 0.0);
    vec3 diffuse = (diff * vec3(mixedTexture)) * light.diffuse;

    // Specular
    vec3 viewDir = normalize(viewPos - FragPos);
    vec3 reflectDir = reflect(-lightDir, norm);
    float spec = pow(max(dot(viewDir, reflectDir), 0.0), material.shininess);
    vec3 specular = mix((vec3(mixedTexture) * spec) * light.specular, light.specular * spec * texture(material.specular, TexCoord).rgb, float(viewSpecular));

    // Combine results
    vec3 result = ambient + diffuse + specular;
    FragColor = vec4(result, 1.0);
}
